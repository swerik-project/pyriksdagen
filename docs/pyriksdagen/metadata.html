 19def increase_date_precision(date, start=True):
 20    if pd.isna(date):
 21        return date
 22    # Year
 23    if len(date) == 4 and start:
 24        return date + '-01-01'
 25    if len(date) == 4 and not start:
 26        return date + '-12-31'
 27    # Month
 28    if len(date) == 7 and start:
 29        return date + '-01'
 30    if len(date) == 7 and start:
 31        last_day = calendar.monthrange(int(date[0]), int(date[1]))[1]
 32        return date + f'-{last_day}'
 33    # Day
 34    if len(date) == 10:
 35        return date

 38def check_date_overlap(start1, end1, start2, end2):
 39    latest_start = max(start1, start2)
 40    earliest_end = min(end1, end2)
 41    delta = (earliest_end - latest_start).days + 1
 42    overlap = max(0, delta)
 43    return True if overlap > 0 else False

226def impute_party(db, party):
227    if 'party' not in db.columns:
228        db['party'] = pd.Series(dtype=str)
229    data = []
230    for i, row in db[db['party'].isnull()].iterrows():
231        parties = party[party['person_id'] == row['person_id']]
232        if len(set(parties['party'])) == 1:
233            db.loc[i,'party'] = parties['party'].iloc[0]
234        if len(set(parties['party'])) >= 2:
235            for j, sow in parties.iterrows():
236                try:
237                    res = check_date_overlap(row['start'], sow['start'], row['end'], sow['end'])
238                except:
239                    LOGGER.error("Impute dates on Corpus using impute_date() before imputing parties!\n")
240                    raise
241                if res:
242                    m = row.copy()
243                    m['party'] = sow['party']
244                    data.append(m)
245    db = pd.concat([db, pd.DataFrame(data)]).reset_index(drop=True)
246    return db

499    """

501        parties = df["party"].unique()
502        if len(parties) == 1:

506        else:

510        return None
511
512    df = corpus.loc[(corpus["person_id"] == person_id) & (pd.notnull(corpus["party"]))].copy()
513    if date is not None:
514        df_date = df.loc[(df["start"] <= date) & (df["end"] >= date)]
515        parties = _party(df_date)
516        if parties is not None:
